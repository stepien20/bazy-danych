    create trigger aktualizacjaWydarzenia after update on memo
    for each row 
    begin
        if new.zrobione>0 THEN
        INSERT INTO sukcesy (wydarzenie, czas)
        VALUES (NEW.tytul, NOW());
end IF;
    end $$
